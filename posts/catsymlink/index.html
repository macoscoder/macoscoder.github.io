读symbol link本身

示例(mac os)

$ go install demo/catsymlink
$ $GOPATH/catsymlink /etc
/private/etc

代码

package main

import (
    "fmt"
    "log"
    "os"
    "path/filepath"
)

func main() {
    if len(os.Args) != 2 {
        fmt.Fprintf(os.Stderr, "Usage: %s link\n", os.Args[0])
        os.Exit(1)
    }
    pth, err := os.Readlink(os.Args[1])
    if err != nil {
        log.Fatal(err)
    }
    pth, err = filepath.Abs(pth)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println(pth)
}

filepath.Abs函数相当于C的realpath函数
